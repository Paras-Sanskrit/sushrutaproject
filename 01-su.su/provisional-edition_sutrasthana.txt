                <head>[Sūtrasthāna]</head>

                <div n="1" type="adhyāyaḥ">
                    <head>[Adhyāya 1]</head>


                    <l xml:id="SS.1.1.0"/>

                    <l xml:id="SS.1.1.1">[1] athāto vedotpattim adhyāyaṃ vyākhyāsyāmaḥ | </l>


                    <l xml:id="SS.1.1.2"/>

                    <l xml:id="SS.1.1.3">[3] atha khalu bhagavantam amaravaram ṛṣigaṇaparivṛttam
                        āśramasthaṃ kāśirājaṃ divodāsam
                        aupadhenavavaitaraṇaurabhrapuṣkalāvatakaravīragopurarakṣitabhojasuśrutaprabhṛtaya
                        ūcuḥ |</l>


                    <l xml:id="SS.1.1.4">[4] bhagavañ śārīramānasāgantubhir vyādhibhir
                        vividhavedanābhighātopadrutān sanāthān anāthavad viceṣṭamānān vikrośataś ca
                        mānavān abhisamīkṣya manasi naḥ pīḍābhavat |</l>

                    <l xml:id="SS.1.1.4b">[4] teṣāṃ sukhaiṣiṇāṃ rogopaśamārtham ātmanaś ca
                        prāṇayātrārtham āyurvedaṃ icchāma upadiśyamānam | atrāyattam aihikam
                        āmuṣmikañ ca śreyas tad bhagavantam upasannāḥ smaḥ śiṣyatveneti |</l>

                    <l xml:id="SS.1.1.5">[5] tān uvāca bhagavān svāgatam vaḥ sarva evāmīmāṃsyā
                        adhyāpyāś ca bhavanto vatsāḥ |</l>

                    <l xml:id="SS.1.1.6">[6] iha khalv āyurvedo nāma <hi>yad</hi> upāṅgam
                        atharvavedasyoktam anutpādyaiva ca prajāḥ ślokaśatasahasram adhyāyasahasrañ
                        ca kṛtavān svayambhūr alpāyuṣkālpamedhastvañ cālokya narāṇām bhūyo 'ṣṭadhā
                        praṇītavān tad yathā | </l>

                    <l xml:id="SS.1.1.7">[7] śalyaṃ śālākyaṃ kāyacikitsābhūtavidyākaumārabhṛtyam
                        agadatantraṃ rasāyanatantraṃ vājīkaraṇatantram iti |</l>

                    <l xml:id="SS.1.1.8.1">[8] athāsya pratyekāṅgalakṣaṇasamāsaḥ |</l> 
                        
                    <l xml:id="SS.1.1.8.1a">[9] tatra śalyan nāma vividhatṛṇakāṣṭhapāṣāṇapāṃsulohaloṣṭāsthibālanakhapūyāsrāvaduṣṭavraṇāntargarbhaśalyoddharaṇārthaṃ
                        yantraśastrakṣārāgnipraṇidhānaviniścayārthañ ca ṣaṣṭyābhidhānair iti | </l>

                    <l xml:id="SS.1.1.8.2">[10] śālākyatantran
                        nāmordhvajatrugatānāṃ vikārāṇāṃ śravaṇanayanavadanaghrāṇādisaṃśritānāṃ
                        vikārāṇām upaśamakaraṇārtham |</l>

                    <l xml:id="SS.1.1.8.3">[11] kāyacikītsā nāma sarvaśarīrāvasthitānāṃ vyādhīnām
                        upaśamakaraṇārthaṃ jvaraśophagulmaraktapittonmādāpasmārapramehātisārādīnāñ
                        ca |</l>

                    <l xml:id="SS.1.1.8.4">[12] bhūtavidyā nāma
                        devagandharvayakṣarākṣasapitṛpiśācavināyakanāgagrahopasṛṣṭacetasāṃ
                        śāntikarmabaliharaṇādigrahopaśamanārthaṃ |</l>
                    
                    <l xml:id="SS.1.1.8.5">[13] kaumārabhṛtyan nāma kumārabharaṇadhātrīkṣīradoṣasaṃśodhanārthaṃ
                        duṣṭastanyagrahasamutthitānāñ ca vyādhīnām upaśamakaraṇārtham |</l>
                    
                    <l xml:id="SS.1.1.8.6">[14] agadatantran nāma sarpakīṭadaṣṭaviṣavyañjanārthaṃ vividhaviṣasaṃyogopaśamanārthañ ca |</l>
                    
                    <l xml:id="SS.1.1.8.7">[15] rasāyanatantran nāma vayaḥsthāpanam āyurmmedhākaraṇaṃ vyādhyupaśamakaraṇārthañ ca|</l>
                    
                    <l xml:id="SS.1.1.8.8">[16] vājīkaraṇatantran nāma
                        svalpaduṣṭakṣīṇaviśuṣkaretasāṃ śukrāpyāyanaprasādopacayajanananimittaṃ
                        praharṣajananārthañ ca |</l>
                    
                    <l xml:id="SS.1.1.9">[17] evam ayam āyurvedo 'ṣṭāṅga upadiśyate atra
                        kasmai kiṃ varṇyatām iti |</l>
                    
                    <l xml:id="SS.1.1.10">[18] ta ūcur asmākaṃ sarvam eva śalyajñānam alaṃ kṛtvopadiśatu bhagavān iti | </l>
                    
                    <l xml:id="SS.1.1.11"> sa uvācaivam astv iti</l>
                    
                    <l xml:id="SS.1.1.12"> ta ūcur bhūyo 'smākaṃ sarveṣām evaikamatīnāṃ matam abhisamīkṣya suśruto bhagavantaṃ prakṣyat<hi>i |</hi> asyopadiśyamānaṃ vayam apy upadhārayiṣyāmaḥ |</l>
                    
                    <l xml:id="SS.1.1.13"> sa uvācaivam astv iti | </l>
