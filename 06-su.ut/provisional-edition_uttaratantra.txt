                    <l xml:id="SS.6.17.1"> [1] athāto dṛṣṭigatarogapratiṣedhaṃ vyākhyāsyāmaḥ ||</l>

                    <l xml:id="SS.6.17.2"/>

                    <l xml:id="SS.6.17.3"> [2] trayaḥ sādhyās trayo 'sādhyāḥ ṣaṭ ca yāpyā bhavanti
                        ha |
                        <caesura/>
                         rogā dṛṣṭigatā nṝṇāṃ teṣu sādhyās trayas traye ||
                        <caesura/>
                         tatraikasya pratīkāraḥ kīrtito dhūmadarśinaḥ | </l>

                    <l xml:id="SS.6.17.4"> [3] dṛṣṭyām pittavidagdhāyāṃ vidagdhāyāṃ kaphena ca |
                        <caesura/>
                         pittaśleṣmaharaṃ kuryād vidhiṃ śastrakṣatād vinā ||</l>

                    <l xml:id="SS.6.17.5"> [4] nasyasekāñjanālepapuṭapākaiḥ satarpaṇaiḥ |
                        <caesura/>
                         ādye tu traiphalaṃ peyaṃ sarpis traivṛtam uttare ||</l>

                    <l xml:id="SS.6.17.6"> [5] tailvakañ cobhayoḥ pathyaṃ kevalaṃ jīrṇam eva vā |
                        <caesura/>
                         gairikaḥ saindhavaḥ kṛṣṇā godantasya maṣī tathā ||</l>

                    <l xml:id="SS.6.17.7"> [6] gomāṃsaṃ maricaṃ bījaṃ śirīṣasya manaḥśilā |
                        <caesura/>
                         vṛntaṃ kapitthān madhunā svayaṃguptāphalāni vā ||</l>

                    <l xml:id="SS.6.17.8"> [7] ete 'ñjane tu catvāro yogāḥ syur ubhayor hitāḥ |
                        <caesura/>
                         kupyakāśokaśālāmrapriyaṅgunalinotpalaiḥ ||</l>

                    <l xml:id="SS.6.17.9"> [8] piṣṭair hareṇvāmalakapathyāpippalisaṃyutaiḥ |
                        <caesura/>
                         kārayed añjanaṃ vaidyaḥ sarpiḥkṣaudrasamāyutaṃ || </l>

                    <l xml:id="SS.6.17.10"> [9] añjayed dvāv api bhiṣak pittasleṣmavibhāvitau |
                        <caesura/>
                         hareṇavo 'mrajambūbhyāṃ puṣpasvarasam eva ca ||</l>

                    <l xml:id="SS.6.17.11"> [10] ghṛtakṣaudravipakvan tu tat prayojyam athāñjanaṃ |
                        <caesura/>
                         nalinotpalakiñjalkagairikā gośakṛdrasaḥ || </l>

                    <l xml:id="SS.6.17.12"> [11] guḍikāñjanam etad vā dinarātryandhayor hitam |
                        <caesura/>
                         rasāñjanaghṛtakṣaudratālīśaṃ svarṇagairikaiḥ || </l>

                    <l xml:id="SS.6.17.13"> [12] gośakṛdrasasaṃyuktaṃ pittopahatadṛṣṭaye |
                        <caesura/>
                         śītasauvīrakam vāpi piṣṭvātha rasabhāvitaṃ || </l>


                    <l xml:id="SS.6.17.14"> [13] kūrmapittena matimān bhāvayed rauhitena vā |
                        <caesura/>
                         cūrṇāñjanena tu sadā prayojyaḥ pittaśāntaye || </l>

                    <l xml:id="SS.6.17.15"> [14] kārśmarī<pc
                        > </pc>puṣpamadhukadārvīlodhrarasāñjanaiḥ |
                        <caesura/>
                         sakṣaudram añjanaṃ tadvā hitam atrāñjane sadā || </l>

                    <l xml:id="SS.6.17.27"> [15] nadīja<pc> </pc>śaṅkha<pc> </pc>trikaṭūny
                        athāñjanaṃ
                        <caesura/>
                         manaḥśilā dve rajane yakṛdrasaḥ |
                        <caesura/>
                         sacandaneyaṃ guḍikāthavāñjane
                        <caesura/>
                         praśasyate vai divaseṣv apaśyatāṃ ||</l>

                    <l xml:id="SS.6.17.16"> [16] srotojaṃ saindhavaṃ kṛṣṇāṃ hareṇuñ cāpi peṣayet |
                        <caesura/>
                         ajāmūtreṇa tā vartyaḥ kṣaṇadāndhāñjane hitāḥ || </l>

                    <l xml:id="SS.6.17.17"> [17] kālānusārivāṃ kṛṣṇāṃ nāgaraṃ madhukan tathā |
                        <caesura/>
                         tālīśapatraṃ rajane saṅkhañ caiva yakṛdrasaḥ || </l>

                    <l xml:id="SS.6.17.18"> [18] piṣṭvā vartyo 'thavā tatra chāyāśuṣkā rujāpahā |
                        <caesura/>
                         manaḥśilābhayāvyoṣaṃ tathā kālānusārivāṃ || </l>

                    <l xml:id="SS.6.17.19"> [19] samudraphenaś ca hitā vartyaś chāgapayonvitāḥ |
                        <caesura/>
                         gomūtrapittamadirāyakṛddhātrīrasaiḥ pacet || </l>

                    <l xml:id="SS.6.17.20"> [20] kṣaudrāñjanaṃ rase vānyadyakṛtas traiphalena vā |
                        <caesura/>
                         gomūtrājyārṇavamalaṃ pippalīkṣaudrakaṭphalaiḥ || </l>

                    <l xml:id="SS.6.17.21"> [21] saindhavopahitaṃ yuñjyān nihitaṃ veṇugahvare |
                        <caesura/>
                         meṣād yakṛd ghṛtañ cājaṃ pippalīsaidhavaṃ madhu || </l>

                    <l xml:id="SS.6.17.22"> [22] rasam āmalakānāñ ca pakvā samyaṅ nidhāpayet |
                        <caesura/>
                         kośake khādire tadvā hitaṃ kṣaudrāñjanaṃ bhavet || </l>

                    <l xml:id="SS.6.17.23"> [23] māgadhyajāsthimajjailāyakṛdyuktā hareṇavaḥ ||
                        <caesura/>

                    <l xml:id="SS.6.17.24">[24] paceta godhāyakṛd antrakalpitaṃ

                        <caesura/>
                         niṣevitaṃ yad yakṛd añjanena vā
                        <caesura/>
                         nihanti naktāndhyam asaṃśayaṃ khalu || </l>
