                        <caesura/>

                        |</l>


                    <l xml:id="SS.6.4.7">[6] śyāvāḥ syuḥ piśitanibhāś ca bindavo ye śuklābhāḥ
                        sitaniyatāḥ sa śuklisañjñaḥ ||
                        <caesura/>
                         eko yaḥ śaśarūdhiropamaś ca binduḥ śuklastho bhavati tam arjanam vadanti
                        |</l>


                    <l xml:id="SS.6.4.8">[7] utsannaḥ salilanibho tha piṣṭaśuklo bindur yaḥ sa
                        bhavati piṣṭakaḥ suvṛttaḥ ||
                        <caesura/>
                         jālābhaḥ kaṭhinasiro mahāṃ saraktaḥ santānaḥ smṛta iha jālasaṃjñitas tu
                        |</l>


                    <l xml:id="SS.6.4.9">[8] śuklasthāḥ sitapiḍākāḥ śirāvṛtāyā yās tāṃ
