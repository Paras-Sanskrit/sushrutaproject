                    <l xml:id="SS.4.2.1"> athātaḥ sadyovraṇacikitsitaṃ vyākhyāsyāmaḥ || </l>

                    <l xml:id="SS.4.2.2"/>

                    <l xml:id="SS.4.2.3"> dhanvantarir ddharmmabhṛtām variṣṭho vāgvidām varaḥ |
                        <caesura/>
                         viśvāmitrātmajam ṛṣiṃ śiṣyaṃ suśrutam anvaśāt || </l>

                    <l xml:id="SS.4.2.4"> nānādhārāmukhaiḥ śastrair nnānāsthānanipātitaiḥ |
                        <caesura/>
                         bhavaṃti nānākṛtayo vraṇās tām̐s tvanibodha me || </l>

                    <l xml:id="SS.4.2.5"> āyatāś caturasrāś ca tryasrā muṇḍalinas tathā |
                        <caesura/>
                         arddhacandrapratīkāśā viśālāḥ kuṭilās tathā || </l>

                    <l xml:id="SS.4.2.6"> śarāvanimnamadhyāś ca yavamadhyās tathāpare |
                        <caesura/>
                         evam prakārākṛtayo bhavanty āgantavo vraṇāḥ || </l>

                    <l xml:id="SS.4.2.7"> doṣajā vā svayambhinnā na tu vaidyanimittajāḥ |
                        <caesura/>
                         vraṇākṛtijño hi bhiṣak na moham upagacchati || </l>

                    <l xml:id="SS.4.2.8"> durddarśaneṣv api bhṛśaṃ vraṇeṣu vikṛteṣv api |
                        <caesura/>
                         anekākṛtir āgantuḥ sa bhiṣagbhiḥ purātanaiḥ || </l>

                    <l xml:id="SS.4.2.9"> samāsato lakṣaṇataḥ ṣaḍvidhaḥ parikīrttitaḥ |
                        <caesura/>
                         chinnam bhinnan tathā viddhaṃ kṣatam picitam eva ca || </l>

                    <l xml:id="SS.4.2.10"> ghṛṣṭam āhus tathā ṣaṣṭhan teṣām vakṣyāmi lakṣaṇaṃ ||
                        <caesura/>
                         tiryakcīno mṛdur vvāpi so vraṇas tv āyato bhavet | </l>

                    <l xml:id="SS.4.2.11"> gātrapātī na vā tad dhi chinnalakṣaṇam ucyate ||
                        <caesura/>
                         śaktikunteṣukhaḍgādiviṣāṇair āśayor hataḥ | </l>

                    <l xml:id="SS.4.2.12"> yat kiñcit sravate tad dhi bhinnalakṣaṇam ucyate ||
                        <caesura/>
                         sthānāny āmāgnipakvānām mūtrasya rudhirasya ca | </l>

                    <l xml:id="SS.4.2.13"> hṛduṇḍukaḥ phupphusaś ca koṣṭham ity abhidhīyate ||
                        <caesura/>
                         tasmin bhinne raktapūrṇṇe jvaro dāhaś ca jāyate | </l>

                    <l xml:id="SS.4.2.14"> mūtramārggagudāsyebhyo raktaṃ ghrāṇāc ca gacchati ||
                        <caesura/>
                         mūrcchādhmānatṛṣāśvāsas tv abhaktacchanda eva ca | </l>

                    <l xml:id="SS.4.2.15"> viṇmūtravātasaṅgañ ca svedāsrāvo kṣiraktatā ||
                        <caesura/>
                         lohagandhitvam āsyasya gātradaurggandham eva ca | </l>

                    <l xml:id="SS.4.2.16"> hṛcchūlaṃ pārśvayoś cāpi viśeṣaṃ cātra me śṛṇu ||
                        <caesura/>
                         āmāśayasthe rudhire rudhiraṃ charddayaty api | </l>

                    <l xml:id="SS.4.2.17"> ādhmānam atimātraṃ ca śūlaṃ ca bhṛśadāruṇam ||
                        <caesura/>
                         pakvāsayagate cāpi rujā gauravam eva ca | </l>

                    <l xml:id="SS.4.2.18"> adhaḥkāye viśeṣeṇa śītatā ca bhavanti hi ||</l>

                    <l xml:id="SS.4.2.19"> sūkṣmāsyaṃ śalyābhihataṃ yadagandhāsayam vinā | </l>

                    <l xml:id="SS.4.2.20"> uttuṇḍitan nirgatam vā tadviddham iti nirddiśet ||
                        <caesura/>
                         nāticchinnan nātividdham ubhayor llakṣaṇānvitam | </l>

                    <l xml:id="SS.4.2.21"> viṣamavraṇasaṃgo yaṃ
                        <caesura/>
                         prahārapīḍanābhyān tu yadaṅgam pṛthutāṃ gataṃ | </l>

                    <l xml:id="SS.4.2.22"> sāsthi tat piccitam vidyāt majjāraktapariplutaṃ ||
                        <caesura/>
                         gharṣaṇād abhighātād vā yad aṅgam vigatatvacaṃ | </l>

                    <l xml:id="SS.4.2.23"> ūṣāsrāvānvitan tan tu ghṛṣṭam ity abhidhīyate ||
                        <caesura/>
                         chinne bhinne tathā viddhe kṣate vāsṛgatisrave | </l>

                    <l xml:id="SS.4.2.24"> raktakṣayād atra rujāḥ karoti pavano bhṛśaṃ || snehapānaṃ
                        hitan tatra tatseko vihitas tathā | </l>

                    <l xml:id="SS.4.2.25"><label>4.2.25</label>veśavāraiḥ sakṛśaraiḥ susnigdhaś
                        copanāhataṃ |
                        <caesura/>
                         dhānyasvedāṃś ca kurvīta snigdhānyālepanāni ca ||</l>


                    <l xml:id="SS.4.2.26"><label>4.2.26</label>vātaghnauṣadhasiddhaiś ca snehair
                        vastir vidhīyate ||
                        <caesura/>
                         piccite ca vighṛṣṭe ca nātisravati śoṇitaṃ |</l>


                    <l xml:id="SS.4.2.27"><label>4.2.27</label>tasmin na gachati bhṛśaṃ dāhaḥ pākaś
                        ca jāyate |
                        <caesura/>
                         tasyoṣmaṇo nigrahārthaṃ dāhapākabhayā ya ca |</l>


                    <l xml:id="SS.4.2.28"><label>4.2.28</label>śītamālepanaṃ kāryam pariṣekaś
                        śītalaḥ |
                        <caesura/>
                         ṣaṭsveteṣu yathokteṣu chinnādiṣu samāsataḥ |</l>


                    <l xml:id="SS.4.2.29"><label>4.2.29</label>jñeyaṃ samarpitaṃ sarvaṃ
                        sadyovraṇacikitsitam || ❈ ||
                        <caesura/>
                         ata ūrdhvam pravakṣyāmi chinnānān tu cikitsitaṃ |</l>


                    <l xml:id="SS.4.2.30"><label>4.2.30</label>ye vraṇā vivṛtāḥ kecic chinnāḥ
                        pārśvāvalamvitaḥ |
                        <caesura/>
                         tāṃ sīvyed vidhinoktena vandhīyād gāḍham eva ca |</l>


                    <l xml:id="SS.4.2.31"><label>4.2.31</label>karṇṇaṃ sthānādapoḍhantu sthāpayutvā
                        yathāsthitaṃ |
                        <caesura/>
                         sīvyedyathoktaṃ tailena srotaś cāpyabhitarpayet |</l>


                    <l xml:id="SS.4.2.32"><label>4.2.32</label>kṛkāṭikānte chinne tu gachaty api
                        samīraṇe |
                        <caesura/>
                         samyaṅ niveśya vadhnīyāt sīvyec cāpi nirantaram |</l>


                    <l xml:id="SS.4.2.33"><label>4.2.33</label>ājena sarpiṣā caivaṃ pariṣeko
                        bhidhīyate |
                        <caesura/>
                         uttānonnaṃ samaśrnīyāc chayīta ca suyantritaḥ |</l>


                    <l xml:id="SS.4.2.34"><label>4.2.34</label>śākhāsu tiryaksatitāṃ prahārāṃ
                        vivṛtāṃ bhṛsaṃ |
                        <caesura/>
                         sīvyet samyaṅ niveśyāśuḥ sandhyasthīnyanupūrvaśaḥ |</l>


                    <l xml:id="SS.4.2.35"><label>4.2.35</label>vadhyāvellitakenāśuḥ vastāntena
                        ghanena vā |
                        <caesura/>
                         carmaṇā goḥ phaṇo vandhaḥ kāryaś vā sahate sinā |</l>


                    <l xml:id="SS.4.2.36"><label>4.2.36</label>pṛṣṭhe vraṇo yasya bhaved uttānañ ca
                        śayīta ca |
                        <caesura/>
                         atonyathā corasije śayīta puruṣo vraṇe |</l>


                    <l xml:id="SS.4.2.37"><label>4.2.37</label>chinnāṃ niḥśeṣataḥ śākhān dagdhvā
                        tailena vuddhimāṃ |
                        <caesura/>
                         vadhnīyāt kośavandhena prāptaṃ kuryāc ca ropaṇaṃ ||</l>


                    <l xml:id="SS.4.2.38"><label>4.2.38</label>candanaṃ padmakaṃ lodhram utpalādi

                        <caesura/>

                    <l xml:id="SS.4.2.39"><label>4.2.39</label>tailam ebhir vipakkaṃ syāt pradhānaṃ

                    <l xml:id="SS.4.2.40"><label>4.2.40</label>hareṇavo mṛṇālañ ca triphalā

                        <caesura/>

                    <l xml:id="SS.4.2.41"><label>4.2.41</label>tailaṃ vipakvaṃ sekāryahitantu

                    <l xml:id="SS.4.2.42"><label>4.2.42</label>bhinnanetramakarmaṇyamabhinnaṃ
