               <l xml:id="SS.4.2.24">hitan tatra tatseko vihitas tathā | </l>


               <l xml:id="SS.4.2.25">veśavāraiḥ sakṛśaraiḥ susnigdhaś copanāhataṃ |
                  <caesura/>
                   dhānyasvedāṃś ca kurvīta snigdhānyālepanāni ca ||</l>


               <l xml:id="SS.4.2.26">vātaghnauṣadhasiddhaiś ca snehair vastir vidhīyate ||
                  <caesura/>
                   piccite ca vighṛṣṭe ca nātisravati śoṇitaṃ |</l>


               <l xml:id="SS.4.2.27">tasmin na gachati bhṛśaṃ dāhaḥ pākaś ca jāyate |
                  <caesura/>
                   tasyoṣmaṇo nigrahārthaṃ dāhapākabhayā ya ca |</l>


               <l xml:id="SS.4.2.28">śītamālepanaṃ kāryam pariṣekaś śītalaḥ |
                  <caesura/>
                   ṣaṭsveteṣu yathokteṣu chinnādiṣu samāsataḥ |</l>


               <l xml:id="SS.4.2.29">jñeyaṃ samarpitaṃ sarvaṃ sadyovraṇacikitsitam || ❈ ||
                  <caesura/>
                   ata ūrdhvam pravakṣyāmi chinnānān tu cikitsitaṃ |</l>


               <l xml:id="SS.4.2.30">ye vraṇā vivṛtāḥ kecic chinnāḥ pārśvāvalamvitaḥ |
                  <caesura/>
                   tāṃ sīvyed vidhinoktena vandhīyād gāḍham eva ca |</l>


               <l xml:id="SS.4.2.31">karṇṇaṃ sthānādapoḍhantu sthāpayutvā yathāsthitaṃ |
                  <caesura/>
                   sīvyedyathoktaṃ tailena srotaś cāpyabhitarpayet |</l>


               <l xml:id="SS.4.2.32">kṛkāṭikānte chinne tu gachaty api samīraṇe |
                  <caesura/>
                   samyaṅ niveśya vadhnīyāt sīvyec cāpi nirantaram |</l>


               <l xml:id="SS.4.2.33">ājena sarpiṣā caivaṃ pariṣeko bhidhīyate |
                  <caesura/>
                   uttānonnaṃ samaśrnīyāc chayīta ca suyantritaḥ |</l>


               <l xml:id="SS.4.2.34">śākhāsu tiryaksatitāṃ prahārāṃ vivṛtāṃ bhṛsaṃ |
                  <caesura/>
                   sīvyet samyaṅ niveśyāśuḥ sandhyasthīnyanupūrvaśaḥ |</l>


               <l xml:id="SS.4.2.35">vadhyāvellitakenāśuḥ vastāntena ghanena vā |
                  <caesura/>
                   carmaṇā goḥ phaṇo vandhaḥ kāryaś vā sahate sinā |</l>


               <l xml:id="SS.4.2.36">pṛṣṭhe vraṇo yasya bhaved uttānañ ca śayīta ca |
                  <caesura/>
                   atonyathā corasije śayīta puruṣo vraṇe |</l>


               <l xml:id="SS.4.2.37">chinnāṃ niḥśeṣataḥ śākhān dagdhvā tailena vuddhimāṃ |
                  <caesura/>
                   vadhnīyāt kośavandhena prāptaṃ kuryāc ca ropaṇaṃ ||</l>


               <l xml:id="SS.4.2.38">candanaṃ padmakaṃ lodhram utpalādi priyaṃgavaḥ ||
                  <caesura/>
                   haridrā madhukaṃ caiva payaś cāṣṭamam eva tu |</l>


               <l xml:id="SS.4.2.39">tailam ebhir vipakkaṃ syāt pradhānaṃ vraṇaropaṇaṃ |
                  <caesura/>
                   candanaṃ karkaṭākhyā ca sahe māṃsāhvayāmṛtā |</l>


               <l xml:id="SS.4.2.40">hareṇavo mṛṇālañ ca triphalā padmakotpalaiḥ |
                  <caesura/>
                   trayodaśāṃgatrivṛtametadvā payasām ānvitaṃ |</l>


               <l xml:id="SS.4.2.41">tailaṃ vipakvaṃ sekāryahitantu vraṇaropane || ❈ ||
                  <caesura/>
                   ata ūrdhvaṃ pravakṣyāmi bhinnānān tu cikitsitaṃ |</l>


               <l xml:id="SS.4.2.42">bhinnanetramakarmaṇyamabhinnaṃ lamvate tu yaḥ |
                  <caesura/>
                   tanniveśya yathāsthānamavyāviddhasiraṃ śanaiḥ |</l>


               <l xml:id="SS.4.2.43">pīḍayen pāṇinā sammyak padmapatrāntareṇa tu |
                  <caesura/>
                   tatosya tarpaṇaṃ kāryan nasyañ cānena sarpiṣā |</l>


               <l xml:id="SS.4.2.44">ajāghṛtaṃ kṣīrapātraṃ madhukañ cotpalāni ca |
                  <caesura/>
                   jīvakarsaṣabhakau cāpi piṣṭvā sarpirvipācayet |</l>


               <l xml:id="SS.4.2.45">sarvanetrābhighāteṣu sarpiretat praśasyate |
                  <caesura/>
                   udarānmedaso varttirn nirgatā yasya dehinaḥ |</l>


               <l xml:id="SS.4.2.46">tāṃ kaṣāyāvakīrṇṇan tu vadhvā sūtreṇa sūtravit |
                  <caesura/>
                   samyag chindyād vraṇena tac ca kṣaudreṇa <unclear extent="1"/> veṣṭayet |</l>

               <l xml:id="SS.4.2.47"/>

               <l xml:id="SS.4.2.48"/>


               <l xml:id="SS.4.2.49">acchindyamānām araṇaṃ kuryād āṭopam eva vā |
                  <caesura/>
                   medāgranthau tu yat tailaṃ vakṣyate tat tu yojayet |</l>


               <l xml:id="SS.4.2.50">tvacotītya sirādīni bhitvā vā parihṛtya vā |
                  <caesura/>
                   kāṣṭhe pratiṣṭhitaṃ śalyaṃ kuryād uktānupadravān |</l>


               <l xml:id="SS.4.2.51">tatrāntarlohitam pāṇḍuśītapādakarānanaṃ |
                  <caesura/>
                   śītocchvāsaṃ raktanetramānaddhañ ca vivarjayet |</l>


               <l xml:id="SS.4.2.52">āmāśayasthe rudhire vamanam pathyam ucyate |
                  <caesura/>
                   pakvāśayathe deyañ ca virecanam asaṃśayaṃ |</l>


               <l xml:id="SS.4.2.53">āsthāpañ ca niḥ snehaḥ kāryam uṣṇaviśodhanaiḥ |
                  <caesura/>
                   yavakolakulatthānāṃ niḥ snehena rasena ca |</l>


               <l xml:id="SS.4.2.54">bhuñjītānnaṃ yavāguṃ vā pivet sendhavasaṃyutāṃ |
                  <caesura/>
                   atiniḥ srutarakto vā bhinnakoṣṭhaḥ pivedsṛk |</l>


               <l xml:id="SS.4.2.55">svamārpratipannās tu yasya viṇmūtramārutāḥ |
                  <caesura/>
                   nirūpadravaḥ sambhinnepi koṣṭhe jīvati mānavaḥ |</l>


               <l xml:id="SS.4.2.56">abhinnamantraṃ niḥ krāntāṃ praveśyaṃ nānyathā bhavet |
                  <caesura/>
                   alaṅgale sirograstaṃ tad apy ake vadanti hi |</l>


               <l xml:id="SS.4.2.57">prakṣālya payasā digdhaṃ tṛṇaśoṇitapāṃsubhiḥ ||
                  <caesura/>
                   praveśayet kṛttanakho ghṛtenāktaṃ śanaiḥ śanaiḥ ||</l>


               <l xml:id="SS.4.2.58">praveśayet kṣīrasiktaṃ śuṣkamantraṃ ghṛtāplataṃ |
                  <caesura/>
                   aṃgulyābhimṛśet kaṇṭhaṃ jalenodvejayed api |</l>


               <l xml:id="SS.4.2.59"/>


               <l xml:id="SS.4.2.60">tathāntrāṇi viśantyantaḥ svāṃ kalāmpīḍayanti ca |
                  <caesura/>
                   vraṇasaudvahutvādvā duḥ praveśantu yad bhavet |</l>


               <l xml:id="SS.4.2.61">tad vipādya pramāṇena bhiṣagantampraveśayet |
                  <caesura/>
                   yathāsthānaniviṣṭe ca vraṇaṃ sīvyamatandritaḥ |</l>


               <l xml:id="SS.4.2.62">sthānādapetamādatte prāṇāṃ gupitam eva vā |
                  <caesura/>
                   veṣṭayitvā ca paṭṭena ghṛtasekampradāpayet |</l>


               <l xml:id="SS.4.2.63">payaḥ pivet sukhoṣṇañ ca citrātailasamanvitaṃ |
                  <caesura/>
                   mṛdukriyārthaśakṛto vāyoścādhaḥ pravṛttaye |</l>


               <l xml:id="SS.4.2.64">tatastailam idaṃ kuryādropaṇārthañ cikitsakaḥ |
                  <caesura/>
                   tvacośvakarṇadhavayorpātakīmeṣaśṛṅgayoḥ |</l>


               <l xml:id="SS.4.2.65">śallakyarjanayoś cāpi vidāryākṣīriṇān tathā |
                  <caesura/>
                   valāmūlāni cāhṛtya tailam etad vipācayet |</l>


               <l xml:id="SS.4.2.66">vraṇaṃ saṃropayet tena raktetsamvatsarañ ca tat ||
                  <caesura/>
                   pādau nirastamuṣkasya jalena praokṣya cākṣiṇī |</l>


               <l xml:id="SS.4.2.67">praveśya muṣkau sūtreṇa tu no sevanasevitaṃ |
                  <caesura/>
                   kāryañ ca goḥ phalāvandhaḥ kaṭyāmāveśya yantrakaṃ |</l>


               <l xml:id="SS.4.2.68"> na kuryāt snehasekañ ca tena praklidyate vraṇaḥ |
                  <caesura/>
                   kālānusārīmagurūṃ candanandevadārū ca |</l>


               <l xml:id="SS.4.2.69">manaḥ śilāle vā hṛtya tailaṃ kurvīta ropaṇaṃ |
                  <caesura/>
                   śīrṣādyapahate śalye vālavartim praveśayet |</l>


               <l xml:id="SS.4.2.70">vālavartyāmadattāyāṃ mastuluṃ<subst>
                     <add place="lower-margin">ga</add>
                  </subst> vraṇāt sravet |
                  <caesura/>
                   hanyādenan tato vāyustasmādevam upācaret |</l>


               <l xml:id="SS.4.2.71">vraṇe rūhati caikaikaṃ śanair vālam upakṣipet |
                  <caesura/>
                   gātrādyapahṛte tasmiṃ snehavartim praveśayet |</l>


               <l xml:id="SS.4.2.72">niḥ śoṇite cāpi kṛtavidhiḥ sadyakṣate hitaḥ |
                  <caesura/>
                   dūrāvagāḍhāḥ sūkṣmā syur ye vraṇāṃstāṃ viśoṇitāṃ |</l>


               <l xml:id="SS.4.2.73">kṛtvā netreṇa sūkṣmeṇa cakratailena tarpayet |
                  <caesura/>
                   padmāṃ samaṅgāṃ rajanīṃ trivargaṃ tuttham eva ca |</l>

               <l xml:id="SS.4.2.74">saṃhṛtya vipacet kāle tailaṃ ropaṇam uttamaṃ |</l>

               <l xml:id="SS.4.2.75"/>

               <l xml:id="SS.4.2.76">kṣate kṣatavidhiḥ kāryaḥ piccite bhagnavad vidhiḥ |</l>

               <l xml:id="SS.4.2.77">ghṛṣṭe rujāṃ nigṛhyāśus cūrṇair upacared vraṇaṃ |
                  <caesura/>
                   viśliṣṭadehaṃ patitaṃ mathitaṃ hatam eva vā | </l>
